Routing

Core component of SPAs

Only one HTML document is delivered, the rest is rendered by JavaScript on client side with the help of templates and data

Routing in AngularJS

HTML:

            
<a href="/page1">page 1</a>
<a href="/page2">page 2</a>

<ng-view></ng-view>

Script:

            
app = angular.module "routing", ["ngRoute"]

app.config [
  "$locationProvider",
  "$routeProvider",
  ($locationProvider, $routeProvider) ->
    $locationProvider.html5Mode true
    $routeProvider
      .when "/page1",
        templateUrl: "/templates/page1.html"
      .when "/page2",
        templateUrl: "/templates/page2.html"
      .otherwise "/page1"
]
