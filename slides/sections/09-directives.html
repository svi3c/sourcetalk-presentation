Custom Drectives

JavaScript:

                
                    angular.module('myApp', [])
                        .controller('myCtrl', function($scope) {
                            $scope.bunny = {
                                name: 'Bugs'
                            };
                        });

                    angular.module('myApp', [])
                        .directive('bunny', function() {
                            return {
                                restrict: 'A',
                                template: 'Name: {{bunny.name}}'
                            };
                        });

HTML:

                
                    <!doctype html>
                    <html ng-app="myApp">
                        <head>
                            <title>Angular Demo</title>
                            <script src="angular.js"></script>
                            <script src="myApp.js"></script>
                            <script src="myCtrl.js"></script>
                            <script src="bunny.js"></script>
                        </head>
                        <body ng-controller="myCtrl">
                            <p bunny></p>
                        </body>
                    </html>

Better Directives

JavaScript:

                
                    angular.module('myApp', [])
                        .directive('bunny', function() {
                            return {
                                restrict: 'A',
                                link: function(scope, element, attributes){
                                    element.html("Name: " + attributes.bunny);
                                }
                            };
                        });

HTML:

                
                    <!doctype html>
                    <html ng-app="myApp">
                        <head>
                            <title>Angular Demo</title>
                            <script src="angular.js"></script>
                            <script src="myApp.js"></script>
                            <script src="myCtrl.js"></script>
                            <script src="bunny.js"></script>
                        </head>
                        <body ng-controller="myCtrl">
                            <p bunny="{{bunny.name}}"></p>
                        </body>
